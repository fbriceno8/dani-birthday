    <script>

        let currentScreen = 0;

        const screens = ['screen1', 'screen2', 'screen3', 'screen4'];

        const timings = [2000, 3000, 5000, 5000]; // Extended final screen from 2s to 5s

        let audioContext;

       

        // Initialize audio context

        function initAudio() {

            if (!audioContext) {

                audioContext = new (window.AudioContext || window.webkitAudioContext)();

            }

            return audioContext;

        }

       

        // Create retro arcade sounds

        function playSound(type, frequency = 440, duration = 0.2) {

            const ctx = initAudio();

            const oscillator = ctx.createOscillator();

            const gainNode = ctx.createGain();

           

            oscillator.connect(gainNode);

            gainNode.connect(ctx.destination);

           

            switch(type) {

                case 'coin':

                    // Classic coin insert sound

                    oscillator.frequency.setValueAtTime(988, ctx.currentTime);

                    oscillator.frequency.setValueAtTime(1319, ctx.currentTime + 0.1);

                    gainNode.gain.setValueAtTime(0.3, ctx.currentTime);

                    gainNode.gain.exponentialRampToValueAtTime(0.001, ctx.currentTime + 0.3);

                    oscillator.type = 'square';

                    duration = 0.3;

                    break;

                   

                case 'powerup':

                    // Achievement unlock sound

                    oscillator.frequency.setValueAtTime(261, ctx.currentTime);

                    oscillator.frequency.setValueAtTime(329, ctx.currentTime + 0.1);

                    oscillator.frequency.setValueAtTime(392, ctx.currentTime + 0.2);

                    oscillator.frequency.setValueAtTime(523, ctx.currentTime + 0.3);

                    gainNode.gain.setValueAtTime(0.2, ctx.currentTime);

                    gainNode.gain.exponentialRampToValueAtTime(0.001, ctx.currentTime + 0.5);

                    oscillator.type = 'square';

                    duration = 0.5;

                    break;

                   

                case 'blip':

                    // Text reveal/screen transition sound

                    oscillator.frequency.setValueAtTime(800, ctx.currentTime);

                    oscillator.frequency.setValueAtTime(600, ctx.currentTime + 0.05);

                    gainNode.gain.setValueAtTime(0.15, ctx.currentTime);

                    gainNode.gain.exponentialRampToValueAtTime(0.001, ctx.currentTime + 0.1);

                    oscillator.type = 'square';

                    duration = 0.1;

                    break;

                   

                case 'startbeep':

                    // Final screen button sound

                    oscillator.frequency.setValueAtTime(1000, ctx.currentTime);

                    gainNode.gain.setValueAtTime(0.25, ctx.currentTime);

                    gainNode.gain.exponentialRampToValueAtTime(0.001, ctx.currentTime + 0.15);

                    oscillator.type = 'square';

                    duration = 0.15;

                    break;

                   

                case 'heartbeat':

                    // Heart beat sound

                    oscillator.frequency.setValueAtTime(200, ctx.currentTime);

                    oscillator.frequency.setValueAtTime(150, ctx.currentTime + 0.05);

                    gainNode.gain.setValueAtTime(0.1, ctx.currentTime);

                    gainNode.gain.exponentialRampToValueAtTime(0.001, ctx.currentTime + 0.1);

                    oscillator.type = 'sine';

                    duration = 0.1;

                    break;

            }

           

            oscillator.start(ctx.currentTime);

            oscillator.stop(ctx.currentTime + duration);

        }

       

        function showScreen(screenIndex) {

            // Hide all screens

            document.querySelectorAll('.screen').forEach(screen => {

                screen.classList.remove('active');

            });

           

            // Show current screen with appropriate sound

            document.getElementById(screens[screenIndex]).classList.add('active');

           

            // Play sounds based on screen

            switch(screenIndex) {

                case 0:

                    // Achievement unlocked - coin + powerup

                    playSound('coin');

                    setTimeout(() => playSound('powerup'), 200);

                    break;

                case 1:

                    // Quest info - blip sound

                    playSound('blip');

                    break;

                case 2:

                    // Love message - gentle blip

                    playSound('blip', 600, 0.15);

                    // Add subtle sounds for each message line appearing

                    setTimeout(() => playSound('blip', 500, 0.1), 500);

                    setTimeout(() => playSound('blip', 550, 0.1), 1000);

                    setTimeout(() => playSound('blip', 500, 0.1), 1500);

                    setTimeout(() => playSound('blip', 550, 0.1), 2000);

                    setTimeout(() => playSound('blip', 600, 0.1), 2500);

                    break;

                case 3:

                    // Final screen - start beep + heartbeat

                    playSound('startbeep');

                    setTimeout(() => playSound('heartbeat'), 500);

                    break;

            }

        }

       

        function startAnimation() {

            showScreen(currentScreen);

           

            setTimeout(() => {

                currentScreen = (currentScreen + 1) % screens.length;

                startAnimation();

            }, timings[currentScreen]);

        }

       

        // Start the animation with user interaction (required for audio)

        function begin() {

            initAudio();

            startAnimation();

            document.getElementById('startButton').style.display = 'none';

        }

       

        // Random screen flicker effect with sound

        setInterval(() => {

            if (Math.random() < 0.05) {

                document.querySelector('.arcade-screen').style.filter = 'brightness(1.2) contrast(1.5)';

                if (Math.random() < 0.3) {

                    playSound('blip', 400, 0.05);

                }

                setTimeout(() => {

                    document.querySelector('.arcade-screen').style.filter = 'none';

                }, 50);

            }

        }, 1000);

       

        // Add heartbeat sound to the heart animation

        setInterval(() => {

            if (currentScreen === 3) {

                playSound('heartbeat');

            }

        }, 1200);
